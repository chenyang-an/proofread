\begin{exercise} Describe Turing machines by drawing a
state diagram and specifying which states are halting,
accepting, rejecting, or output states.
\begin{enumerate}
\item Prove that the set $\{ 1^n0 : n\ge 0 \} = \{0, 10, 110, 1110, 11110, \ldots \}$
is Turing decidable by giving a Turing machine that decides it.
\item
Prove that the set $\{ 1^n0 : n\ge 0 \} = \{0, 10, 110, 1110, 11110, \ldots \}$
is Turing enumerable by giving a Turing machine that enumerates it.
\end{enumerate}
\end{exercise}

\begin{exercise}
Let the input alphabet have $k$ symbols $\Sigma = \{ s_1, \ldots s_k \}$.
Describe a Turing machine~$M$ that decides the
set
\[
R ~=~ \{w \in \Sigma^* :
        \hbox{$w$ is nonempty and has its first symbol the same as the last symbol} \} .
\]
How many states does your machine~$M$ have as a function of~$k$?
\end{exercise}

\begin{exercise}
Let the input alphabet be $\Sigma = \{0,1 \}$. Give the state diagram
for a Turing machine that computes the string concatenation
function $\langle u,v \rangle \mapsto\penalty10000 uv$. Describe the input
and output conditions. Where is the tape head placed in
the halting configuration? What are the initial configurations in the cases
where one or both of $u$ or~$v$ are the empty string?
\end{exercise}

\begin{exercise}\label{exer:CompareTM}
Give the explicit state diagram of a Turing machine~$M_{=}$ that computes
the binary equality relation. Namely, $M_{=}$~accepts
the pair $\langle v, w \rangle$ if $v=w$, and rejects
if $v \neq w$. [Hint: You might wish to use the breadcrumb
technique, similar to what was done for copying a string
in Example~\ref{example:CopyW}; this is especially useful if you
want to carry out the comparison without destroying
the values of $v$ and~$w$.
It is also possible to non-destructively compare
$v$ and~$w$ without using breadcrumbs.]
\end{exercise}

\begin{exercise}
Recall that $w^\TR$ is the reversal of~$w$. Prove the following by
giving an explicit description of a Turing machine, preferably in the form of
a state diagram. (One way to do these is with the breadcrumb technique.
There are Turing
machines with ten states that work, but we do not know if ten states is optimal.)
\begin{enumerate}
\item[(a)] Prove that the \idxnm{reversal} function
          $w \mapsto w^\TR$ is Turing computable.
\item[(b)] Prove that $\{ w : \hbox{$w\in\Sigma^*$ and $w=w^\TR$} \}$ is Turing decidable.  This is the set of palindromes\index{palindrome}.
\end{enumerate}
\end{exercise}

\begin{exercise}\label{exer:binaryUnary}
Give high-level descriptions
of how Turing machines with
input and tape alphabets
$\Sigma = \Gamma = \{0,1,\#\}$ can convert between unary and binary
representations for integers, so that the following
functions are computable:
\begin{enumerate}
\item The function $1^n \mapsto \Tstr(n)$,
\item The function $w \mapsto 1^n$ where $w$~is a binary representation of~$n$.
\end{enumerate}
\end{exercise}

\begin{exercise}\label{exer:symboldoubling}
Give the state diagrams for Turing machines that use only the
tape symbols $\Gamma = \{0, 1, \#\}$ and compute
the following functions:
\begin{enumerate}
\item
The \idxnm{symbol-doubling} function
$a_1 a_2 \cdots a_{n-1} a_n \mapsto a_1 a_1 a_2 a_2 \cdots a_{n-1} a_{n-1} a_n a_n$;
so that $0$'s are replaced with ``$00$'', and $1$'s with ``$11$''.
\item The symbol-halving function
$a_1 a_2 a_3 a_4 \cdots a_{n-1} a_n \mapsto a_2 a_4 a_6 \cdots a_{2 \lfloor n/2\rfloor}$
that computes an inverse to the symbol-doubling function.
\end{enumerate}
\end{exercise}

\begin{exercise}
\ ~
\begin{enumerate}
\item Prove that there is a Turing machine~$M$ such that: (i)~If $M$ is started
with the input
tape is completely blank, then $M$ does not halt, and
(ii)~If $M$ is started with a non-blank
symbol anywhere on the tape, then $M$ halts.
\item Prove that there is no Turing machine~$N$ such that:
(i)~If $N$ is started
with the input
tape completely blank, then $N$~eventually accepts, and
(ii)~If $N$ is started with a non-blank
symbol anywhere on the tape, then $N$ rejects.
\end{enumerate}
\end{exercise}

\begin{exercise}
For this exercise, restrict attention to Turing machines
that use the alphabets $\Sigma = \Pi = \{\#,1\}$ and partial
compute unary functions.
Prove that it is undecidable whether a given Turing machine
has the least possible number of states. That is, let $X$ be the
binary relation
\[
X = \{\langle \lgn M\rgn, n \rangle : \hbox{\parbox[t]{4in}{\raggedright
   $M$~is a Turing machine and there is no Turing machine~$M^\prime$ \newline
       \hspace*{1em} that partial computes the same partial
               function as~$M$ \newline
       \hspace*{1em} and has at most~$n$ states.\,\}}}
\]
Prove $X$ is undecidable. As a consequence, there is no algorithm to
minimize the number of states in a Turing machine.
\end{exercise}

\begin{exercise}
(The Busy Beaver function for running time.) For this exercise and the
next one, restrict attention
to Turing Machines over the alphabets $\Sigma = \Gamma = \{1,\#\}$.
With this restriction, there are only finitely many Turing machines
with $n$ states (up to renaming of states).
Define the
running-time version of
the Busy Beaver function $\TBBsteps: \mathbb N \rightarrow \mathbb N$ by
\[
\TBBsteps(n) ~=~ \max \{ m \ge 0 : \hbox{\parbox[t]{3in}{\raggedright
    there is a Turing Machine~$M$ with $n$ states \newline
    \hspace*{2ex}such that $M(\epsilon)$ halts after exactly $m$ steps.~$\}$}}
\]
\begin{enumerate}
\item Prove that $\TBBsteps(n)$ is not computable by proving that otherwise
$\THalt^\TTM_0$ would be decidable.  (Here $\THalt^\TTM_0$ means the version
of $\THalt_0$ where G{\"o}del numbers of algorithms are given in the
form of Turing machines. All the theorems proved 
in Chapter~\ref{chap:AlgorithmsInformal} for $\THalt_0$ apply
also to $\THalt_0^\TTM$.)
\item Suppose $f: \mathbb N \rightarrow \mathbb N$ is computable. Prove that $\TBBsteps(n)$
eventually dominates~$f$, namely there is an~$N$ such that $f(n)<\TBBsteps(n)$ for all
$n>N$.
\end{enumerate}
\end{exercise}

\begin{exercise}(Busy Beaver function for output value.)
Use the same conventions as the previous exercise.
The Busy Beaver function $\TBB: \mathbb N \rightarrow \mathbb N$
is defined by
\[
\TBB(n) ~=~ \max \{ m \ge 0 : \hbox{\parbox[t]{3.5in}{\raggedright
    There is a Turing Machine~$M$ with $n$ states such \newline
    \hspace*{2ex}that $M(\epsilon)$ halts with $m$ many~$1$'s written on the tape.~$\}$}}
\]
Suppose $f: \mathbb N \rightarrow \mathbb N$ is computable. Prove that $\TBB(n)$
eventually dominates~$f$. Conclude that $\TBB$ is not computable.
\end{exercise}
