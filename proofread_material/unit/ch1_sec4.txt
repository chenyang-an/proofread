\section{Satisfiability, Tautologies, and Implication}
\label{sec:SatContra}\label{sec:satProp}

\paragraph*{Satisfiability and tautologies.}
A formula is a ``\idxnm{tautology}'' if it is always true; it is ``\idxnm{satisfiable}''
if it can sometimes be true.  Formally:

\begin{definition}
A formula~$A$ is a \idxem{tautology} if $\varphi(A) = \True$
for all truth assignments~$\varphi$. In this case, we also
say that $A$ is \idxem{tautologically valid}.

A formula~$A$ is \idxem{satisfiable} if there is some
truth assignment~$\varphi$ such that $\varphi(A) = \True$.
In this case, we say that $\varphi$ is a \idxem{satisfying assignment} for~$A$
or that $A$ is \idxem{satisfied by}~$\varphi$.
If there is no satisfying assignment for~$A$, then $A$~is
\idxem{unsatisfiable}.
\end{definition}

\noindent
The notation ``$\vDash A$'' will also be used to
denote that $A$~is a tautology. For this, see
Definition~\ref{def:vDash} below.

\begin{example}
The formula $p_1$ is satisfiable but not a tautology, since $\varphi(p_1)$
may equal either $\True$ or~$\False$.  The formulas $p_1 \lor \lnot p_1$
and $p_1\limplies (p_2\limplies p_1)$ are tautologies, and hence are satisfiable.
The formula $p_1\land \lnot p_1$ is unsatisfiable.

As already discussed, Figure~\ref{fig:truthpOqpNq} shows
that $p \land q \limplies p\lor q$ is a tautology
and that $p \lor q \limplies p\land q$ is satisfiable but not a tautology.
\end{example}

It may not be immediately obvious that the formula
$p_1 \limplies p_2 \limplies p_1$ is a tautology;
one way to check this is by the method of
truth tables\index{truth table}. Figure~\ref{fig:p1p2p1} shows the truth
table, which shows that $p_1 \limplies p_2 \limplies p_1$
is true under all four possible
combinations of values of
$\varphi(p_1)$ and $\varphi(p_2)$.

More generally, suppose a formula~$A$ involves
$k$~many distinct variables, $p_{i_1},\ldots,p_{i_k}$.
There are $2^k$ many ways to set the values
of $\varphi(p_{i_j})$. A truth table for~$A$ will thus
have $2^k$ lines. This gives an algorithm for
determining whether a given formula $A$ is
satisfiable and/or a tautology; namely, consider all
possible $2^k$ truth assignments~$\varphi$ on the variables in~$A$
and evaluate $\varphi(A)$.
Then $A$ is a tautology if and only if $\varphi(A) = \True$ for all $2^k$ assignments.
And $A$ is satisfiable if and only if $\varphi(A) = \True$ for at least one
of the truth assignments.
\bigskip

Now let $\Gamma$ denote a set of propositional
formulas. It is often interesting to know
whether there is a truth assignment that
makes every formula in~$\Gamma$ true.

\begin{definition}\label{def:satisfiableProp}
A set $\Gamma$ of propositional formulas
is \idxem{satisfiable} if there is a truth assignment
$\varphi$ such that $\varphi(A) = \True$ for
every $A\in \Gamma$.  For such a $\varphi$, we
say that $\varphi$ \idxem{satisfies}~$\Gamma$ or
that $\varphi$~is a \idxem{satisfying assignment} for~$\Gamma$
or that $\Gamma$~is \idxem{satisfied by}~$\varphi$.
\end{definition}

The set~$\Gamma$ may be finite or infinite.
If $\Gamma$~is finite, $\Gamma = \{A_1, A_2,\ldots, A_k\}$,
then $\Gamma$ is satisfied by~$\varphi$ if and only
if $\varphi$~satisfies $A_1 \land A_2 \land \cdots \land A_k$.
Whether this holds can be determined by the method of truth tables.

Thus, if $\Gamma$ is finite, the method of truth tables
can be used to determine whether $\Gamma$ is satisfiable.
However, the method of truth tables cannot be
used if $\Gamma$ is infinite. This is because, in general, there
may be infinitely many variables in~$\Gamma$, and one would
have to consider infinitely many truth assignments.

\begin{example}\label{example:GammaSat}
Let $\Gamma$ be the set of formulas
\[
\Gamma ~=~ \{ p_i\liff \lnot p_{i+1} : i\ge 1 \}
       ~=~ \{ p_1 \liff \lnot p_2,\, p_2 \liff \lnot p_3,\, p_3 \liff \lnot p_4,\, \ldots \}.
\]
Note that $p_i\liff \lnot p_{i+1}$ is satisfied by~$\varphi$ exactly when $\varphi$~assigns
opposite truth values to $p_i$ and~$p_{i+1}$.
Therefore, there are exactly two satisfying assignments for~$\Gamma$. The first
satisfying assignment,~$\varphi_1$, has $\varphi_1(p_{2j-1}) =\True$ and
$\varphi_1(p_{2j}) = \False$ for all $j\ge 1$.  The second one,~$\varphi_2$,
has  $\varphi_2(p_{2j-1}) =\penalty10000 \False$ and
$\varphi_2(p_{2j}) = \True$ for all $j\ge 1$.
\end{example}

\begin{example}\label{example:GammaSat2}
Let $\Gamma'$ be the set of formulas
\[
\Gamma' ~=~ \{ p_i\limplies p_{i+1} : i\ge 1 \}
        ~=~ \{ p_1 \limplies p_2,\, p_2 \limplies p_3,\, p_3 \limplies p_4,\, \ldots \}.
\]
For each $i\ge 0$, there is a satisfying assignment~$\varphi_i$ such that
$\varphi_i(p_j) = \True$ for $j\ge i$ and $\varphi_i(p_j) = \False$ for $j<i$.
Thus, there are infinitely many satisfying assignments.

These are the only satisfying assignments for~$\Gamma'$.
\end{example}

\paragraph*{Tautological implication.}
A \idxem{tautological implication} is an implication that
is true under any truth assignment. This is defined
formally as follows:

\begin{definition}\label{def:vDash}
Let $A$ and~$B$ be formulas, and $\Gamma$ be a set of formulas.
\begin{enumerateLoose}
\item We say that $A$~\idxem{tautologically implies}~$B$,
or just $A$ \idxem{implies}~$B$ for short, provided that
every truth assignment that satisfies~$A$ also satisfies~$B$.
We write $A \vDash B$ to denote that $A$~tautologically implies~$B$.
\item We say that $\Gamma$~\idxem{tautologically implies}~$B$,
or just $\Gamma$ \idxem{implies}~$B$ for short, provided that
every truth assignment that satisfies~$\Gamma$ also satisfies~$B$.
We write $\Gamma \vDash B$ to denote that $\Gamma$~tautologically implies~$B$.
\end{enumerateLoose}
\end{definition}

Unwinding Definitions \ref{def:satisfiableProp} and~\ref{def:vDash}(b),
$\Gamma \vDash A$ means the same as
\[
\hbox{For all~$\varphi$, if $\varphi(B)=\True$ for all $B\in \Gamma$, then $\varphi(A)= \True$.}
\]
Or equivalently,
\[
\hbox{For all~$\varphi$, if $\varphi$ satisfies $\Gamma$, then $\varphi$ satisfies $A$.}
\]

The symbol ``$\vDash$'' is called the
``double turnstile''\index{double turnstile@{double turnstile ($\vDash$)}} sign.
Note that $A \vDash B$ means the same thing as $\{A \} \vDash B$.
It is common to take liberties in notation and write things like
$A,B \vDash C$ for $\{  A, B \} \vDash C$,
and $\Gamma, A \vDash B$ for $\Gamma \cup \{ A \} \vDash B$.
In all cases, it means that any truth assignment that satisfies
all the formulas and sets of formulas to the left of the $\vDash$ sign
also satisfies the formula to the right of the $\vDash$ sign.
Finally, we write just $\vDash A$ for $\varnothing \vDash A$, where
$\varnothing$ is the empty set (of no formulas).  The condition
$\vDash A$ holds if and only if $A$~is a tautology.

We write $\Gamma \nvDash A$ to mean that $\Gamma \vDash A$ is
false.

\begin{example}
Some examples of tautological (non)-implication with a finite set~$\Gamma$ include:
\begin{itemizeLoose}
\item[(a)] $p_1, p_1\limplies p_2 \vDash p_2$.
\item[(b)] $p_1, p_2  \vDash p_2 \land p_1$.
\item[(c)] $p_1 \nvDash p_2$.
\item[(d)] $\vDash p_1 \limplies p_2 \limplies p_1$.
\item[(e)] $\nvDash (p_1 \limplies p_2) \limplies p_1$.
\end{itemizeLoose}
These can be verified by using truth tables. Note that (a) means the
same as $\{p_1, p_1\limplies p_2\} \vDash p_2$, and similarly for~(b).
\end{example}

\begin{example}\label{example:GammaSatRevisited}
Let $\Gamma$ be the set of formulas from Example~\ref{example:GammaSat},
and $\varphi_1$ and~$\varphi_2$ be its two satisfying assignments.
Then
\begin{itemizeLoose}
\item[(a)] $\Gamma \nvDash p_1$ and $\Gamma \nvDash \lnot p_1$.
\item[(b)] $\Gamma \vDash p_1\liff p_3$.
\end{itemizeLoose}
Condition~(a) holds since
$\varphi_2(p_1)=\False$ and $\varphi_1(\lnot p_1) = \False$.
Condition~(b) holds since $\varphi_1$ and~$\varphi_2$
are the only satisfying
assignments, and since $\varphi_i(p_1) =\varphi_i(p_3)$ and
thus $\varphi_i(p_1\liff p_3) = \True$
for both $i=1$ and $i=2$.
\end{example}

\begin{theorem}\label{thm:subsetGammaDelta}
Suppose that $\Gamma$ and~$\Delta$ are sets of formulas,
and $\Gamma \subseteq \Delta$.
\begin{enumerate}
\item If $\varphi$ satisfies~$\Delta$, then $\varphi$ satisfies~$\Gamma$.
\item If $\Delta$ is satisfiable, then $\Gamma$~is also satisfiable.
\item If $\Gamma \vDash A$, then $\Delta \vDash A$.
\end{enumerate}
\end{theorem}

Theorem~\ref{thm:subsetGammaDelta} is an immediate consequence of the
definitions. \hfill $\qed$

\paragraph*{Implication from an unsatisfiable set.}
An unsatisfiable set~$\Gamma$ tautologically implies all formulas~$B$:

\begin{theorem}\label{thm:AandNotA}
\ ~~
\begin{itemize}
\item[(a)] Let $A$ and~$B$ be formulas. Then $A,\lnot A \vDash B$.
\item[(b)] Let $B$ be a formula and $\Gamma$ be an unsatisfiable set of formulas.
Then $\Gamma \vDash B$.
\end{itemize}
\end{theorem}

Part~(a) is a special case of part~(b) since
$\{ A, \lnot A\}$ is unsatisfiable. Part~(b) is proved
by observing that since there is no truth assignment
satisfying~$\Gamma$, the definition of tautological
implication means that $\Gamma\vDash B$ holds
trivially.
\hfill $\qed$

\paragraph*{Tautologically equivalent formulas.}
Two formulas are
``tautologically equivalent\index{tautologically equivalent@{tautologically equivalent ($\tautEq$)}}''
if they tautologically imply each other:

\begin{definition}
We write $A\tautEq B$ to denote that both $A\vDash B$
and $B\vDash A$.  In this case, we say that $A$ and~$B$ are
\emph{tautologically equivalent}\index{tautologically equivalent@{tautologically equivalent ($\tautEq$)}}.
Immediately from the
definitions, $A\tautEq B$ holds if and only
if $\varphi(A)=\varphi(B)$ for all truth assignments.
\end{definition}

\paragraph*{The semantic deduction theorem.}

The next theorem is the semantic version of what will
later be called the Deduction Theorem
(see Section~\ref{sec:propDeductionThm}).

\begin{theorem}[Semantic Deduction Theorem]\label{thm:deductionSemanticProp}%
~~\ \newline\index{Deduction Theorem!semantic}%
Let $A$ and $B$ be formulas
and $\Gamma$~be a set of formulas. Then,
\begin{itemizeLoose}
\item[(a)] $A\vDash B$ if and only if $\vDash A \limplies B$.
\item[(b)] $\Gamma , A \vDash B$ if and only if $\Gamma \vDash A \limplies B$.
\end{itemizeLoose}
\end{theorem}

\begin{proof}
Part~(a) is the special case of~(b) with $\Gamma$ equal to the empty
set. So let's prove~(b). The proof is basically just
unwinding definitions. To start, the condition $\Gamma, A \vDash B$ means that
for any truth assignment $\varphi$, if $\varphi$ satisfies $\Gamma \cup \{A\}$,
then $\varphi(B) = \True$.  Equivalently, this means that
if $\varphi$ satisfies~$\Gamma$ and $\varphi(A) = \True$, then $\varphi(B) = \True$.
By the definition of truth for~$\limplies$, the condition
\[
\hbox{ ``if $\varphi(A) = \True$, then $\varphi(B) = \True$'' }
\]
is equivalent to $\varphi(A\limplies B) = \True$.  Therefore,
$\Gamma, A \vDash B$ is equivalent to the condition that
for all $\varphi$, if $\varphi$ satisfies~$\Gamma$,
then $\varphi(A\limplies B) = \True$. In other words, it is
equivalent to $\Gamma \vDash A \limplies B$.
\end{proof}

\begin{example}\label{example:GammaSat3}
Returning to the set $\Gamma$ of 
Examples \ref{example:GammaSat} and~\ref{example:GammaSatRevisited},
we have $\Gamma, p_1 \vDash\penalty10000 p_3$ and $\Gamma, p_3 \vDash p_1$.
\end{example}

\begin{theorem}
Suppose $\Gamma$ is a finite set of formulas, $\Gamma = \{A_1,\ldots, A_k\}$.
Then $\Gamma \vDash B$ if and only if $A_1\land A_2 \land \cdots \land A_k \vDash B$.
\end{theorem}

\begin{proof}
A truth assignment
$\varphi$ satisfies~$\Gamma$
if and only if it satisfies $A_1\land A_2 \land \cdots \land A_k$.
Therefore, $\Gamma \vDash B$ if and only if
$A_1\land A_2 \land \cdots \land A_k \vDash B$.
\end{proof}

\begin{theorem}\label{thm:pqrEquivPqr}
$A \tautEq B$ if and only if $\vDash A\liff B$.
\end{theorem}

Theorem~\ref{thm:pqrEquivPqr} follows readily from the
Semantic Deduction Theorem
and the fact that $\vDash A\liff B$ holds if and only if
both $\vDash A\limplies B$ and $\vDash B \limplies A$.

\paragraph*{Duality between satisfiability and validity.}
The next two theorems show how the property of being satisfiable
is dual to the property of being a tautology.

\begin{theorem}\label{thm:dualSatTaut}
Let $A$ be a formula. Then $A$ is a tautology if and only
if $\lnot A$~is not satisfiable.
\end{theorem}

Theorem~\ref{thm:dualSatTaut} is a special case of the next theorem,
with $\Gamma = \varnothing$. This is a precursor to
Theorems \ref{thm:proofbyContra1} and~\ref{thm:proofbyContra2}
about \idxnm{proof by contradiction}.

\begin{theorem}\label{thm:ContradictionSemantic}
Let $\Gamma$ be a set of formulas and $A$~be a formula.
Then $\Gamma \vDash A$ if and only if $\Gamma\cup\{\lnot A\} $
is unsatisfiable.
\end{theorem}
\begin{proof}
The condition that $\Gamma \vDash A$ means that every truth assignment~$\varphi$
that satisfies~$\Gamma$ has $\varphi(A) = \True$. The condition that
$\Gamma \cup\{\lnot A\}$ is unsatisfiable is equivalent to the condition
that every truth assignment that satisfies~$\Gamma$ has $\varphi(\lnot A) = \False$.
These two conditions are clearly equivalent.
\end{proof}
