\section*{Exercises}

\nonumSectionmark{Exercises}

Your answers for Exercises \ref{exer:firstCh3}-\ref{exer:lastBeforeUseSoundness} should
not use the Completeness Theorem.
Exercises \ref{exer:firstCh3}-\ref{exer:lastExplicitPL} can be done
using only the material of Section~\ref{sec:PL}.
Exercises \ref{exer:firstDeduct}-\ref{exer:lastDeduct}
are intended to be done using only the techniques developed up through
Section~\ref{sec:propDeductionThm} (on the Deduction Theorem).

\begin{exercise}\label{exer:firstCh3}
Give explicit $\PL$-proofs for the following formulas. The explicit
proofs should show every line in the $\PL$-proofs.
\begin{enumerate}
\item $A\limplies B \vdash A\limplies B$.
\item $A \limplies B, B \vdash B$.
\item $A \limplies B, A \vdash B$.
\item $ \lnot A, A \vdash B$
\end{enumerate}
\end{exercise}

\begin{exercise}
Show that $\vdash (A\limplies B) \limplies (A\limplies A)$ by giving an explicit
three line $\PL$-proof
of that formula.
\end{exercise}

\begin{exercise}\label{exer:lastExplicitPL}
Recall that $A\land B$ is an abbreviation for $\lnot(A \limplies \lnot B)$.
Give explicit $\PL$-proofs for:
\begin{enumerate}
\item $A\land B, \lnot B \vdash A$
\item $A\land B, \lnot A \vdash B$
\end{enumerate}
\end{exercise}

\begin{exercise}\label{exer:firstDeduct}
Prove the following. You only need to prove that a $\PL$-proof exists; you do not need to give it
explicitly.
\begin{enumerate}
\item $A\limplies B, \, C \vdash A \limplies (C \limplies B)$. Are both hypotheses needed?
\item $A \limplies B \limplies C \vdash  B \limplies A \limplies C$.
\end{enumerate}
\end{exercise}

\begin{exercise}\label{exer:LongHypSyllogism}
~\
\begin{enumerate}
\item Prove $\vdash (A\limplies B) \limplies (B\limplies C) \limplies (C\limplies D ) \limplies (A\limplies D)$.
\item Conclude the following generalization of \idxnm{Hypothetical Syllogism} is
a valid derived rule of inference for~$\PL$:
\begin{prooftree}
\AxiomC{$A\limplies B$}
\AxiomC{$B\limplies C$}
\AxiomC{$C\limplies D$}
\TrinaryInfC{$A\limplies D$}
\end{prooftree}
\end{enumerate}
\end{exercise}

\begin{exercise}
Prove $\vdash (A \lor B) \limplies \lnot A \limplies B$.
\end{exercise}

\begin{exercise}\label{exer:lastDeduct}
Prove $ A\limplies B \vdash (\lnot A \limplies A) \limplies B$.
\end{exercise}

\begin{exercise}\label{exer:byCases}
Show $\vdash (A\limplies B) \limplies (\lnot A \limplies B) \limplies B$.
(Compare with Exercise~\ref{exer:byCasesSemantic}.)
\end{exercise}

\begin{exercise}
Prove that the following are inconsistent.
\begin{enumerate}
\item $\{\lnot (p_1 \limplies p_1)\}$.
\item $\{ p_1 \land\penalty10000 \lnot p_1 \}$.
\end{enumerate}
\end{exercise}

\begin{exercise}
Prove that $\Gamma$ is inconsistent if and only if
$\Gamma \vdash \lnot (p_1 \limplies p_1)$.
\end{exercise}

\begin{exercise}\label{exer:AimpAorB}
Show that the following formulas have $\PL$-proofs. (See Example~\ref{example:implyAorB}.)
\begin{itemize}
\item[(a)] $A \limplies A \lor B$.
\item[(b)] $B \limplies A \lor B$.
\item[(c)] $\lnot(A \lor B) \limplies (\lnot A \land \lnot B)$.
\end{itemize}
\end{exercise}

\begin{exercise}\label{exer:andOrImpPLproofs}
Show that the following formulas have $\PL$-proofs.
\begin{enumerate}
\item $(A \land B) \limplies A$.
\item $(A \land B) \limplies B$.
\item $\lnot(A \land B) \limplies (\lnot A \lor \lnot B)$.
\item $A \limplies B \limplies A\land B$.
\end{enumerate}
\end{exercise}

\begin{exercise}\label{exer:impliesNestProofs}
Do the following for each of the formulas (a)-(c):
(i)~State whether it is unsatisfiable, or satisfiable; 
(ii)~State whether it is a tautology or not. 
(ii)~If it is not a tautology, give a truth assignment that falsifies it;
(iii)~If it is a tautology, prove that it has a $\PL$-proof.
\begin{itemize}
\item[(a)] $p \limplies (q\limplies p) \limplies p$.
\item[(b)] $(p \limplies (q\limplies p)) \limplies p$.
\item[(c)] $((p \limplies q)\limplies p) \limplies p$.
\end{itemize}
\end{exercise}

\begin{exercise}\label{exer:andInconsis}
Prove that $\Gamma\cup \{A \land B\}$ is inconsistent
if and only if $\Gamma\cup\{A,B\}$ is inconsistent.
[Hint: You may use the results of Exercise~\ref{exer:andOrImpPLproofs} in your answer.]
\end{exercise}

\begin{exercise}\label{exer:orInconsis}
Show that $\Gamma, A\lor B$ is inconsistent
if and only if both $\Gamma\cup\{A\}$ and $\Gamma\cup\{B\}$ are inconsistent.
\end{exercise}

\begin{exercise}\label{exer:OrLeft}
Show that $\Gamma , A\lor B \vdash C$ holds if and only
if $\Gamma, A\vdash C$ and $\Gamma, B\vdash C$ both hold. [Hint: Use proof-by-cases.]
\end{exercise}

\begin{exercise}\label{exer:iffInconsis}
Show that $\Gamma, A\liff B$ is inconsistent
if and only if both $\Gamma\cup\{A,B\}$ and $\Gamma\cup\{\lnot A, \lnot B\}$ are inconsistent.
\end{exercise}

\begin{exercise}\label{exer:PiercePL}
Prove that $\vdash ((p\limplies q) \limplies p)\limplies p$.
This tautology is known as \idxnm{Pierce's Law} and is sometimes used
as an axiom related to the \idxnm{Law of the Excluded Middle}. (This is a
duplicate of Exercise~\ref{exer:impliesNestProofs}(c).)
\end{exercise}

\begin{exercise} Show that $\{A\limplies B, A, \lnot B \}$ is
inconsistent. (This was used in the proof of Lemma~\ref{lem:completePLconds}.)
\end{exercise}

\begin{exercise}
Prove
\begin{enumerate}
\item $\{\lnot(A\limplies B), \lnot A\}$ is inconsistent.
\item $\{\lnot(A\limplies B), B\}$ is inconsistent.
\end{enumerate}
These facts were used in the proof of Lemma~\ref{lem:completePLconds}.
\end{exercise}

\begin{exercise}\label{exer:AimpliesBinconsis}
Prove that $\Gamma\cup \{A \limplies B\}$ is inconsistent
if and only if $\Gamma\vdash A$ and $\Gamma\cup\{B\}$ is inconsistent.
(Compare to Theorem~\ref{thm:casesAimpliesB} and to item~(5) in the
list in Section~\ref{sec:constructingPL}.)
\end{exercise}

\begin{exercise}\label{exer:GeneralizedCases}\label{exer:lastBeforeUseSoundness}
(Generalization of \idxnm{proof-by-cases}.)
Suppose that $\Gamma \vdash A\lor B$ and $\Gamma\vdash A \limplies C$
and $\Gamma \vdash B\limplies C$. Prove that $\Gamma \vdash C$.
[Hint: One way to do this is to prove
\[
\vdash (A\limplies C) \limplies (B\limplies C) \limplies (A\lor B \limplies C).
\]
This is the last axiom for~$\lor$ on page~\pageref{eq:lastOrAxiom}
and is related to Exercise~\ref{exer:OrLeft}. 
In any event, it is suggested to use proof-by-cases.]
\end{exercise}

\begin{exercise}\label{exer:completenessBtoA}
Give a direct proof that part~(a) of the Soundness
Theorem~\ref{thm:completePL} implies part~(b).  (See the last part
of the proof of the Soundness Theorem.)
\end{exercise}

\begin{exercise}\label{exer:equivComplete}
Suppose that
$\Gamma \vDash p_i$ or $\Gamma \vDash \lnot p_i$, for every~$i$.
Prove that,
for every formula~$A$, either $\Gamma \vDash A$ or $\Gamma \vDash \lnot A$.
(This is similar to being \idxnm{complete}; however, instead of having
one of $A$ or~$\lnot A$ a member of~$\Gamma$, we have one of
$A$ or~$\lnot A$ tautologically implied by~$\Gamma$.)
\end{exercise}

\begin{exercise}\label{exer:jointlyInconsisPL}
Suppose that $\Gamma$ and $\Delta$ are sets of formulas
and that $\Gamma \cup \Delta$ is unsatisfiable.
\begin{enumerate}
\item Prove that there is a finite $\Gamma^\prime \subseteq \Gamma$
and a finite $\Delta^\prime \subseteq \Delta$ such
that $\Gamma^\prime \cup \Delta^\prime$ is unsatisfiable.
\item Prove that there is a formula~$A$ such that
$\Gamma \vDash A$ and $\Delta \vDash \lnot A$.
\end{enumerate}
\end{exercise}

\begin{exercise}\label{exer:complementary}
Two sets $\Gamma$ and $\Delta$ of formulas are
called \idxem{complementary} if every truth assignment~$\varphi$,
satisfies exactly one of $\Gamma$ and~$\Delta$.
Suppose $\Gamma$ and $\Delta$ are complementary.
Prove there are subsets $\Gamma'\subseteq \Gamma$
and $\Delta' \subseteq \Delta$ such that $\Gamma'$ and~$\Delta'$
are finite and such that $\Gamma' \vDash \Gamma$ and
$\Delta' \vDash \Delta$.
(Recall that the notation $\Gamma' \vDash \Gamma$ means $\Gamma' \vDash A$ for
all $A\in \Gamma$.)
\end{exercise}

\begin{exercise}
(This builds on Exercise~\ref{exer:complementary}.)
\begin{itemize}
\item[(a)] Prove that a set~$\Gamma$ has a complementary set~$\Delta$
if and only if there is a formula~$A$ such that $\Gamma\vDash A$ and
$A\vDash \Gamma$.  
\item[(b)] Give an example of a set~$\Gamma$ which does not
have a complementary set. Prove that it does not have
a complementary set.
\end{itemize}
\end{exercise}

\begin{exercise}\label{exer:3colorableProp}
Use the Compactness Theorem for propositional logic to
prove that a graph~$G$ is 3-colorable\index{colorable@{3-colorable graph}} if and only if every finite
subgraph is 3-colorable. (``3-colorable'' means there is an assignment of three
colors to the vertices of the graph so that no edge connects vertices assigned the
same color.) For this, fix a graph~$G$.
Use propositional variables
$r_i, g_i, b_i$ whose intended meanings are that
``Vertex~$i$ is red'', ``Vertex~$i$ is green'', and ``Vertex~$i$ is blue'', respectively.
Let $\Gamma$ be a set of formulas using these variables that expresses the
conditions that (a)~each vertex has a color assigned to it, and
(b)~if two vertices $i$ and~$j$ are joined by an edge in~$G$, then they
are not assigned the same color. The set~$\Gamma$ should
be satisfiable if and only if $G$ is 3-colorable.
Then apply the Compactness Theorem.
\end{exercise}

\begin{exerciseStar}\label{exer:altProfCompletePL}
Show that the reductions of items (1)-(5) of Section~\ref{sec:constructingPL} can
be used to give an alternate proof of the Completeness Theorem for~$\PL$ by working
out the details of the following proof sketch.

First, define $N(\Gamma)$ to be the number
of occurrences of $\limplies$'s and $\lnot$'s in formulas in~$\Gamma$ minus
the number of negated literals $\lnot p_i$ appearing in~$\Gamma$. In other words,
$N(\Gamma)$ counts only the negation signs~$\lnot$ that are not part of literals.
Prove that the following two statements both hold (use induction on~$n$ for
the second statement):
\begin{quote}
If $N(\Gamma)=0$ and $\Gamma$ is unsatisfiable, then $\Gamma$~is inconsistent.
\newline
If $N(\Gamma)+N(\{A\}) = n$ and $\Gamma \vDash A$, then $\Gamma \vdash A$.
\end{quote}
Each induction step will use one or two of the reductions given in
items (1)-(5) of Section~\ref{sec:constructingPL}.

In the base case, $N(\Gamma)=0$ means that $\Gamma$~is a set
of literals. In this case, show that either $\Gamma$~is satisfiable
or $\Gamma$ is inconsistent. Also show that if $N(\Gamma)=N(\{A\})=0$,
then either $\Gamma\nvDash A$ or $\Gamma \vdash A$.
\end{exerciseStar}

\begin{exerciseStar}\label{exer:uncountCompletenessPL}
(For readers who know some set theory.)
The proof of \idxnm{Lindenbaum's Theorem}~\ref{thm:lindenbaumPL}
depended on the fact that, since there are only countably many variables~$p_i$,
there are only countably many formulas. It is also possible to allow uncountably
many propositional variables $p_\alpha$ where $\alpha \in \mathcal I$ for an
arbitrary set~$\mathcal I$. Use Zorn's Lemma to prove
Lindenbaum's Theorem also holds when there are uncountably many
variables and hence uncountably many formulas. Use this to
prove the Completeness\index{Completeness Theorem}
and Compactness\index{Compactness Theorem} Theorems hold also for propositional
formulas over uncountably many variables.
\end{exerciseStar}

\begin{exerciseStar}\label{exer:compactnessOtherPf}
Give a proof of the \idxnm{Compactness Theorem} that does not depend on the
Soundness or Completeness Theorems. Use the following proof sketch.
\begin{itemize}
\item[(a)] Prove that if $\Gamma$ is finitely satisfiable and $A$ is
a formula, then at least one of $\Gamma \cup \{A\}$ or $\Gamma\cup \{\lnot A\}$
is finitely satisfiable.
\item[(b)] Prove the analogue of \idxnm{Lindenbaum's Theorem} for
finitely satisfiability instead of consistency. Namely, prove that
if $\Gamma$ is finitely satisfiable, then there is a complete,
finitely satisfiable $\Pi \supseteq \Gamma$.
\item[(c)] Prove that the properties of Lemma~\ref{lem:completePLconds} hold
under the assumption that $\Pi$ is complete and finitely satisfiable.
\item[(d)] Prove the analogue of Lemma~\ref{lem:completePL} that states
that if $\Pi$ is complete and finitely satisfiable, then $\Pi$ is
satisfiable.  (Exactly the same proof works for this part with no changes needed.)
\end{itemize}
\end{exerciseStar}
